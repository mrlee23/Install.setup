** Instalar
  :PROPERTIES: 
  :CUSTOM_ID: install
  :LANG: es
  :END:      

*** Ubuntu

#+BEGIN_SRC sh
$ sudo apt-get install es2fsprogs
#+END_SRC

*** CentOS

#+BEGIN_SRC sh
$ sudo yum install es2fsprogs
#+END_SRC

*** from Source code

#+BEGIN_SRC sh
$ git clone https://git.kernel.org/pub/scm/fs/ext2/e2fsprogs.git
$ mkdir build
$ cd build
$ ../configure
$ make
$ make check
$ sudo make install
#+END_SRC
